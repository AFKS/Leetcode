# 204. Count Primes

## Description
> Description:

### Hash Table
```C++
class Solution {
public:
    int countPrimes(int n) {
        vector<bool> isPrime(n,true);
        isPrime[0] = false, isPrime[1] = false;
        for (int i=0; i*i<=n;++i){
            if (isPrime[i]){
                for (int j = i; j*i<=n; j++)
                    isPrime[j*i]=false;
            }
        }
        return count(isPrime.begin(), isPrime.end(), true);
    }
};
```
