### Stack

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isValid(string s) {
        stack<char> stk;
        for (auto it = s.begin(); it!=s.end(); ++it){
            if (stk.empty()) stk.push(*it);
            else{
                if ((stk.top()=='('&&*it==')')||(stk.top()=='['&&*it==']')||(stk.top()=='{'&&*it=='}'))
                    stk.pop();
                else stk.push(*it);
            }
        }
        if (stk.empty()) return true;
        return false;
    }
};
```
